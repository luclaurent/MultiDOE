0025 function foldersLoad=initDirMultiDOE(pathcustom,other)
0026 
0027 % variable 'other' (optional) of type cell must constain the list of other
0028 % toolboxes to load (they must be in '../.')
0029 
0030 % variable 'pathcustom' (optional) contains the specific folder from where
0031 % the directories must be loaded
0032 
0033 %folders of the MultiDOE toolbox
0034 foldersLoad={'src',...
0035     'src/crit',...
0036     'src/disp',...
0037     'src/DOE',...
0038     'src/init',...
0039     'src/libs',...
0040     'src/libs/LHS',...
0041     'src/libs/IHS',...
0042     'src/various',...
0043     'src/libs',...
0044     'src/libs/m2html'};
0045 
0046 %depending on the parameters
0047 specifDir=true;
0048 if nargin==0
0049     specifDir=false;
0050 elseif nargin>1
0051     if isempty(pathcustom)
0052         specifDir=false;
0053     end
0054 end
0055 %if no specified directory
0056 if ~specifDir
0057     pathcustom=strrep(which(mfilename),[mfilename '.m'],'');
0058 end
0059 
0060 %absolute paths
0061 pathAbsolute=cellfun(@(c)fullfile(pathcustom,c),foldersLoad,'uni',false);
0062 
0063 %add to the PATH
0064 flA=cellfun(@(x)addpathExisted(x),pathAbsolute);
0065 flB=[];
0066 if nargin==2
0067     if ~isempty(other)
0068         %Load other toolbox
0069         if ~iscell(other);other={other};end
0070         %absolute paths
0071         pathAbsolute=cellfun(@(c)fullfile(pathcustom,'/../',c),other,'uni',false);
0072         %add to the PATH
0073         cellfun(@(x)addpathExisted(x),pathAbsolute);
0074         %add other toolbox to the PATH
0075         namFun=cellfun(@(c)['initDir' c],other,'uni',false);
0076         flB=cellfun(@feval,namFun,pathAbsolute);
0077     end
0078 end
0079 
0080 if any([flA flB]==2)
0081     %display
0082     Mfprintf(' ## Toolbox: MultiDOE loaded\n');
0083 end
0084 end
