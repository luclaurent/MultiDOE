0022 function foldersLoad=initDirMultiDOE(pathcustom,other)
0023 
0024 % variable 'other' (optional) of type cell must constain the list of other
0025 % toolboxes to load (they must be in '../.')

0027 % variable 'pathcustom' (optional) contains the specific folder from where
0028 % the directories must be loaded

0030 %folders of the MultiDOE toolbox
0031 foldersLoad={'src',...
0032     'src/crit',...
0033     'src/disp',...
0034     'src/DOE',...
0035     'src/init',...
0036     'src/libs',...
0037     'src/libs/LHS',...
0038     'src/libs/IHS',...
0039     'src/various',...
0040     'src/libs',...
0041     'src/libs/m2html'};
0042 
0043 %depending on the parameters
0044 specifDir=true;
0045 if nargin==0
0046     specifDir=false;
0047 elseif nargin>1
0048     if isempty(pathcustom)
0049         specifDir=false;
0050     end
0051 end
0052 %if no specified directory
0053 if ~specifDir
0054     pathcustom=pwd;
0055 end
0056 
0057 %absolute paths
0058 pathAbsolute=cellfun(@(c)[pathcustom '/' c],foldersLoad,'uni',false);
0059 
0060 %add to the PATH
0061 cellfun(@addpath,pathAbsolute);
0062 
0063 if nargin==2
0064     %Load other toolbox
0065     if ~iscell(other);other={other};end
0066     %absolute paths
0067     pathAbsolute=cellfun(@(c)[pathcustom '/../' c],other,'uni',false);
0068     %add to the PATH
0069     cellfun(@addpath,pathAbsolute);
0070     %add other toolbox to the PATH
0071     namFun=cellfun(@(c)['initDir' c],other,'uni',false);
0072     cellfun(@feval,namFun,pathAbsolute)
0073 end
0074 
0075 %display
0076 fprintf(' ## Toolbox: MultiDOE loaded\n');
