0020 function [distMat,distVect,uniqDist,multiDOE] = calcDist(sampling,q)
0021 
0022 %if q is not specified, the euclidian distance will be used
0023 if nargin==1
0024     q=2;
0025 end
0026 
0027 %number of variables
0028 ns=size(sampling,1);
0029 
0030 %generate combinations
0031 %comb=fullfact([ns ns]);
0032 comb=allcomb(1:ns,1:ns); % replace fullfact if not available
0033 comb1=comb(:,2);comb2=comb(:,1); % switch 1 and 2 in the case of fullfact
0034 diffC=comb1-comb2;
0035 iX=diffC~=0;
0036 iXX=diffC<0;
0037 
0038 iX=iX&iXX;
0039 
0040 comb1=comb1(iX);
0041 comb2=comb2(iX);
0042 %[comb1 comb2]
0043 %compute inter-sample points distances
0044 pti=sampling(comb1,:);
0045 ptj=sampling(comb2,:);
0046 distVect=sum(abs(ptj-pti).^q,2).^(1/q);
0047 %store in a matrix
0048 distMat=zeros(ns);
0049 IX=sub2ind([ns ns],comb1,comb2);
0050 distMat(IX)=distVect;
0051 distMat=distMat+distMat';
0052 
0053 if nargout>1
0054     %concatenate distance
0055     dist_cat=distVect(:);    
0056     if nargout>2
0057         %remove duplicates
0058         uniqDist=unique(dist_cat);
0059         if nargout>3
0060             %build table of the multiplicity of the distances
0061             multiDOE=zeros(size(uniqDist));
0062             for ii=1:length(uniqDist)
0063                 multiDOE(ii)=sum(ismember(dist_cat,uniqDist(ii)));
0064             end
0065         end
0066     end
0067 end
