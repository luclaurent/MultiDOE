0022 %% function for computing inter-sample distances
0023 %% L. LAURENT -- 03/04/2013 -- luc.laurent@lecnam.net
0024 
0025 function [distMat,distVect,uniqDist,multiDOE] = calcDist(sampling,q)
0026 
0027 %if q is not specified, the euclidian distance will be used
0028 if nargin==1
0029     q=2;
0030 end
0031 
0032 %number of variables
0033 ns=size(sampling,1);
0034 
0035 %generate combinations
0036 %comb=fullfact([ns ns]);
0037 comb=allcomb(1:ns,1:ns); % replace fullfact if not available
0038 comb1=comb(:,2);comb2=comb(:,1); % switch 1 and 2 in the case of fullfact
0039 diffC=comb1-comb2;
0040 iX=diffC~=0;
0041 iXX=diffC<0;
0042 
0043 iX=iX&iXX;
0044 
0045 comb1=comb1(iX);
0046 comb2=comb2(iX);
0047 %[comb1 comb2]
0048 %compute inter-sample points distances
0049 pti=sampling(comb1,:);
0050 ptj=sampling(comb2,:);
0051 distVect=sum(abs(ptj-pti).^q,2).^(1/q);
0052 %store in a matrix
0053 distMat=zeros(ns);
0054 IX=sub2ind([ns ns],comb1,comb2);
0055 distMat(IX)=distVect;
0056 distMat=distMat+distMat';
0057 
0058 if nargout>1
0059     %concatenate distance
0060     dist_cat=distVect(:);    
0061     if nargout>2
0062         %remove duplicates
0063         uniqDist=unique(dist_cat);
0064         if nargout>3
0065             %build table of the multiplicity of the distances
0066             multiDOE=zeros(size(uniqDist));
0067             for ii=1:length(uniqDist)
0068                 multiDOE(ii)=sum(ismember(dist_cat,uniqDist(ii)));
0069             end
0070         end
0071     end
0072 end
