0020 function [distMat,distVect,uniqDist,multiDOE] = calcDist(sampling,q)
0021 
0022 %if q is not specified, the euclidian distance will be used
0023 if nargin==1
0024     q=2;
0025 end
0026 
0027 %number of variables
0028 ns=size(sampling,1);
0029 
0030 %geenrate combinations
0031 comb=fullfact([ns ns]);
0032 comb1=comb(:,1);comb2=comb(:,2);
0033 diffC=comb1-comb2;
0034 iX=diffC~=0;
0035 iXX=diffC<0;
0036 
0037 iX=iX&iXX;
0038 
0039 comb1=comb1(iX);
0040 comb2=comb2(iX);
0041 %[comb1 comb2]
0042 %compute inter-sample points distances
0043 pti=sampling(comb1,:);
0044 ptj=sampling(comb2,:);
0045 distVect=sum(abs(ptj-pti).^q,2).^(1/q);
0046 %store in a matrix
0047 distMat=zeros(ns);
0048 IX=sub2ind([ns ns],comb1,comb2);
0049 distMat(IX)=distVect;
0050 distMat=distMat+distMat';
0051 
0052 if nargout>1
0053     %concatenate distance
0054     dist_cat=distVect(:);    
0055     if nargout>2
0056         %remove duplicates
0057         uniqDist=unique(dist_cat);
0058         if nargout>3
0059             %build table of the multiplicity of the distances
0060             multiDOE=zeros(size(uniqDist));
0061             for ii=1:length(uniqDist)
0062                 multiDOE(ii)=sum(ismember(dist_cat,uniqDist(ii)));
0063             end
0064         end
0065     end
0066 end
