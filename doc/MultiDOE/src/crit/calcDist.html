0033 %% function for computing inter-sample distances
0034 %% L. LAURENT -- 03/04/2013 -- luc.laurent@lecnam.net
0035 
0036 function [distMat,distVect,uniqDist,multiDOE] = calcDist(sampling,q)
0037 
0038 %if q is not specified, the euclidian distance will be used
0039 if nargin==1
0040     q=2;
0041 end
0042 
0043 %number of variables
0044 ns=size(sampling,1);
0045 
0046 %geenrate combinations
0047 comb=fullfact([ns ns]);
0048 comb1=comb(:,1);comb2=comb(:,2);
0049 diffC=comb1-comb2;
0050 iX=diffC~=0;
0051 iXX=diffC<0;
0052 
0053 iX=iX&iXX;
0054 
0055 comb1=comb1(iX);
0056 comb2=comb2(iX);
0057 %[comb1 comb2]
0058 %compute inter-sample points distances
0059 pti=sampling(comb1,:);
0060 ptj=sampling(comb2,:);
0061 distVect=sum(abs(ptj-pti).^q,2).^(1/q);
0062 %store in a matrix
0063 distMat=zeros(ns);
0064 IX=sub2ind([ns ns],comb1,comb2);
0065 distMat(IX)=distVect;
0066 distMat=distMat+distMat';
0067 
0068 if nargout>1
0069     %concatenate distance
0070     dist_cat=distVect(:);    
0071     if nargout>2
0072         %remove duplicates
0073         uniqDist=unique(dist_cat);
0074         if nargout>3
0075             %build table of the multiplicity of the distances
0076             multiDOE=zeros(size(uniqDist));
0077             for ii=1:length(uniqDist)
0078                 multiDOE(ii)=sum(ismember(dist_cat,uniqDist(ii)));
0079             end
0080         end
0081     end
