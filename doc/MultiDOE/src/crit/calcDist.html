0018 %% function for computing inter-sample distances
0019 %% L. LAURENT -- 03/04/2013 -- luc.laurent@lecnam.net
0020 
0021 function [distMat,distVect,uniqDist,multiDOE] = calcDist(sampling,q)
0022 
0023 %if q is not specified, the euclidian distance will be used
0024 if nargin==1
0025     q=2;
0026 end
0027 
0028 %number of variables
0029 ns=size(sampling,1);
0030 
0031 %generate combinations
0032 %comb=fullfact([ns ns]);
0033 comb=allcomb(1:ns,1:ns); % replace fullfact if not available
0034 comb1=comb(:,2);comb2=comb(:,1); % switch 1 and 2 in the case of fullfact
0035 diffC=comb1-comb2;
0036 iX=diffC~=0;
0037 iXX=diffC<0;
0038 
0039 iX=iX&iXX;
0040 
0041 comb1=comb1(iX);
0042 comb2=comb2(iX);
0043 %[comb1 comb2]
0044 %compute inter-sample points distances
0045 pti=sampling(comb1,:);
0046 ptj=sampling(comb2,:);
0047 distVect=sum(abs(ptj-pti).^q,2).^(1/q);
0048 %store in a matrix
0049 distMat=zeros(ns);
0050 IX=sub2ind([ns ns],comb1,comb2);
0051 distMat(IX)=distVect;
0052 distMat=distMat+distMat';
0053 
0054 if nargout>1
0055     %concatenate distance
0056     dist_cat=distVect(:);    
0057     if nargout>2
0058         %remove duplicates
0059         uniqDist=unique(dist_cat);
0060         if nargout>3
0061             %build table of the multiplicity of the distances
0062             multiDOE=zeros(size(uniqDist));
0063             for ii=1:length(uniqDist)
0064                 multiDOE(ii)=sum(ismember(dist_cat,uniqDist(ii)));
0065             end
0066         end
0067     end
0068 end
