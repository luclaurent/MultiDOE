0022 %% Computation of the score associated to a sampling
0023 % L. LAURENT -- 19/12/2012 -- luc.laurent@lecnam.net
0024 
0025 function [uniform,discrepancy]=calcScore(sampling,q)
0026 
0027 % norme use for computing the distance
0028 if nargin==1
0029     q=2;
0030 end
0031 
0032 %number of points and variables
0033 ns=size(sampling,1);
0034 np=size(sampling,2);
0035 
0036 %the sampling is reduced to the unit hypercube (space [0,1]^d)
0037 minSampling=min(sampling);
0038 maxSampling=max(sampling);
0039 p=1./(maxSampling-minSampling);
0040 c=-minSampling.*p;
0041 samplingN=sampling.*repmat(p,ns,1)+repmat(c,ns,1);
0042 
0043 % Compute inter-sample points distance + multiplicity(see Forrester,Sobester&Keane
0044 % 2008)
0045 [distM,distV,distU,pairS]=calcDist(sampling,q);
0046 %meme calculs sur les valeurs normees
0047 [distNM,distNV,distNU,paireNS]=calcDist(samplingN,q);
0048 
0049 %remove diagonals
0050 distM(logical(eye(ns)))=NaN;
0051 distNM(logical(eye(ns)))=NaN;
0052 %% with raw data
0053 %minimal distance of each sample points
0054 minDistPt=min(distM);
0055 %average minimal distance
0056 minDistAvg=mean(minDistPt);
0057 %minimal distance
0058 uniform.minDist=distU(1);
0059 %sum of the inverse of the distances (see Leary et al. 2004)
0060 uniform.sumDist=sum(1./distU(:).^2);
0061 %%criteria for the PhD thesis of Jessica FRANCO 2008
0062 %average minimal distance
0063 uniform.avgMinDist=minDistAvg;
0064 %%
0065 %£ with normalized data
0066 %minimal distance of each sample points
0067 minDistPtN=min(distNM);
0068 %average minimal distance
0069 minDistAvg=mean(minDistPtN);
0070 %minimal distance (maximin)
0071 uniform.minDistN=distNU(1);
0072 %sum of the inverse of the distances (see Leary et al. 2004)
0073 uniform.sumDistN=sum(1./distNU(:).^2);
0074 %%criteria for the PhD thesis of Jessica FRANCO 2008
0075 %measure of the covering/uniformity
0076 uniform.cover=1/minDistAvg*...
0077     (1/ns*sum(minDistPtN-minDistAvg)^2)^.5;
0078 %ratio of the distances
0079 uniform.distRatio=distNU(end)/distNU(1);
0080 %average minimal distance
0081 uniform.avgMinDistN=minDistAvg;
0082 
0083 %criteria from Morris & Mitechll 1995
0084 uniform.morris=sum(pairS.*distU.^(-q))^(1/q);
0085 uniform.morrisN=sum(paireNS.*distNU.^(-q))^(1/q);
0086 
0087 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
0088 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
0089 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
0090 %% Discrepancy
0091 %(formulas from Fang et al., Uniform Design: Theory and Application
0092 %2000 & and from Jessica Franco)
0093 
0094 if nargout>=2
0095     %value in [0,1]^d
0096     sampling=samplingN;
0097     %initialize
0098     samplingM=reshape(sampling,ns,1,np);
0099     samplingM=repmat(samplingM,[1 ns 1]);
0100     ind=[2 1 3];
0101     samplingMB=permute(samplingM,ind);
0102     samplingMC=samplingM-0.5;
0103     samplingMBC=samplingMB-0.5;
0104     
0105     % L2-discrepancy
0106     part1=prod(1-sampling.^2,2);
0107     maxSampling=max(samplingM,samplingMB);
0108     prodm=prod(1-maxSampling,3);
0109     part2=sum(prodm(:));
0110     %     part2=0;
0111     %     for ii=1:nb_val
0112     %         for jj=1:nb_val
0113     %             tmp=1;
0114     %             for ll=1:nb_var
0115     %                 tmp=tmp*(1-max(tirages(ii,ll),tirages(jj,ll)));
0116     %             end
0117     %             part2=part2+tmp;
0118     %         end
0119     %     end
0120     
0121     discrepancy.L2=3^(-np)-...
0122         2^(1-np)/ns*sum(part1)...
0123         +1/ns*part2;
0124     
0125     % Centered L2-discrepancy
0126     part1=prod(1+0.5*abs(sampling-0.5)-0.5*(sampling-0.5).^2,2);
0127     opt=1+0.5*abs(samplingMC)+0.5*abs(samplingMBC);
0128     prodm=prod(opt,3);
0129     part2=sum(prodm(:));
0130     %     part2=0;
0131     %     for ii=1:nb_val
0132     %         for jj=1:nb_val
0133     %             tmp=1;
0134     %             for ll=1:nb_var
0135     %                 tmp=tmp*(1+0.5*abs(tirages(ii,ll)-0.5)+...
0136     %                     0.5*abs(tirages(jj,ll)-0.5));
0137     %             end
0138     %             part2=part2+tmp;
0139     %         end
0140     %     end
0141     
0142     
0143     discrepancy.CL2=(13/12)^np-2/ns*sum(part1)+...
0144         1/ns^2*part2;
0145     
0146     % Symetric L2-discrepancy
0147     part1=prod(1+2*sampling-2*sampling.^2,2);
0148     opt=1+0.5*abs(samplingMC)+0.5*abs(samplingMBC)-...
0149         0.5*abs(samplingMC-samplingMBC);
0150     prodm=prod(opt,3);
0151     part2=sum(prodm(:));
0152     %     part2=0;
0153     %     for ii=1:nb_val
0154     %         for jj=1:nb_val
0155     %             tmp=1;
0156     %             for ll=1:nb_var
0157     %                 tmp=tmp*(1+0.5*abs(tirages(ii,ll)-0.5)+...
0158     %                     0.5*abs(tirages(jj,ll)-0.5)-...
0159     %                     0.5*abs(tirages(ii,ll)-tirages(jj,ll)));
0160     %             end
0161     %             part2=part2+tmp;
0162     %         end
0163     %     end
0164     
0165     discrepancy.SL2=(4/3)^np-...
0166         2/ns*sum(part1)+...
0167         2^np/ns^2*part2;
0168     
0169     % Modified L2-discrepancy
0170     part1=prod(3-sampling.^2,2);
0171     opt=1-abs(samplingM-samplingMB);
0172     prodm=prod(opt,3);
0173     part2=sum(prodm(:));
0174     %     part2=0;
0175     %     for ii=1:nb_val
0176     %         for jj=1:nb_val
0177     %             tmp=1;
0178     %             for ll=1:nb_var
0179     %                 tmp=tmp*(1-abs(tirages(ii,ll)-tirages(jj,ll)));
0180     %             end
0181     %             part2=part2+tmp;
0182     %         end
0183     %     end
0184     discrepancy.ML2=(4/3)^np-...
0185         2^(1-np)/ns*sum(part1)+...
0186         1/ns^2*part2;
0187     
0188     % wrap around L2-discrepancy
0189     part1=abs(samplingM-samplingMB);
0190     opt=3/2-part1.*(1-part1);
0191     prodm=prod(opt,3);
0192     part2=sum(prodm(:));
0193     discrepancy.WL2=ns*(-(4/3)^np+...
0194         1/ns^2*part2);
0195     
0196     
0197 end
