0018 %% Computation of the score associated to a sampling
0019 % L. LAURENT -- 19/12/2012 -- luc.laurent@lecnam.net

0021 function [uniform,discrepancy]=calcScore(sampling,q)
0022 
0023 % norme use for computing the distance
0024 if nargin==1
0025     q=2;
0026 end
0027 
0028 %number of points and variables
0029 ns=size(sampling,1);
0030 np=size(sampling,2);
0031 
0032 %the sampling is reduced to the unit hypercube (space [0,1]^d)
0033 minSampling=min(sampling);
0034 maxSampling=max(sampling);
0035 p=1./(maxSampling-minSampling);
0036 c=-minSampling.*p;
0037 samplingN=sampling.*repmat(p,ns,1)+repmat(c,ns,1);
0038 
0039 % Compute inter-sample points distance + multiplicity(see Forrester,Sobester&Keane
0040 % 2008)
0041 [distM,distV,distU,pairS]=calcDist(sampling,q);
0042 %meme calculs sur les valeurs normees
0043 [distNM,distNV,distNU,paireNS]=calcDist(samplingN,q);
0044 
0045 %remove diagonals
0046 distM(logical(eye(ns)))=NaN;
0047 distNM(logical(eye(ns)))=NaN;
0048 %% with raw data
0049 %minimal distance of each sample points
0050 minDistPt=min(distM);
0051 %average minimal distance
0052 minDistAvg=mean(minDistPt);
0053 %minimal distance
0054 uniform.minDist=distU(1);
0055 %sum of the inverse of the distances (see Leary et al. 2004)
0056 uniform.sumDist=sum(1./distU(:).^2);
0057 %%criteria for the PhD thesis of Jessica FRANCO 2008
0058 %average minimal distance
0059 uniform.avgMinDist=minDistAvg;
0060 %%
0061 %£ with normalized data
0062 %minimal distance of each sample points
0063 minDistPtN=min(distNM);
0064 %average minimal distance
0065 minDistAvg=mean(minDistPtN);
0066 %minimal distance (maximin)
0067 uniform.minDistN=distNU(1);
0068 %sum of the inverse of the distances (see Leary et al. 2004)
0069 uniform.sumDistN=sum(1./distNU(:).^2);
0070 %%criteria for the PhD thesis of Jessica FRANCO 2008
0071 %measure of the covering/uniformity
0072 uniform.cover=1/minDistAvg*...
0073     (1/ns*sum(minDistPtN-minDistAvg)^2)^.5;
0074 %ratio of the distances
0075 uniform.distRatio=distNU(end)/distNU(1);
0076 %average minimal distance
0077 uniform.avgMinDistN=minDistAvg;
0078 
0079 %criteria from Morris & Mitechll 1995
0080 uniform.morris=sum(pairS.*distU.^(-q))^(1/q);
0081 uniform.morrisN=sum(paireNS.*distNU.^(-q))^(1/q);
0082

0084 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
0085 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
0086 %% Discrepancy
0087 %(formulas from Fang et al., Uniform Design: Theory and Application
0088 %2000 & and from Jessica Franco)

0090 if nargout>=2
0091     %value in [0,1]^d
0092     sampling=samplingN;
0093     %initialize
0094     samplingM=reshape(sampling,ns,1,np);
0095     samplingM=repmat(samplingM,[1 ns 1]);
0096     ind=[2 1 3];
0097     samplingMB=permute(samplingM,ind);
0098     samplingMC=samplingM-0.5;
0099     samplingMBC=samplingMB-0.5;
0100     
0101     % L2-discrepancy
0102     part1=prod(1-sampling.^2,2);
0103     maxSampling=max(samplingM,samplingMB);
0104     prodm=prod(1-maxSampling,3);
0105     part2=sum(prodm(:));
0106     %     part2=0;
0107     %     for ii=1:nb_val
0108     %         for jj=1:nb_val
0109     %             tmp=1;
0110     %             for ll=1:nb_var
0111     %                 tmp=tmp*(1-max(tirages(ii,ll),tirages(jj,ll)));
0112     %             end
0113     %             part2=part2+tmp;
0114     %         end
0115     %     end
0116     
0117     discrepancy.L2=3^(-np)-...
0118         2^(1-np)/ns*sum(part1)...
0119         +1/ns*part2;
0120     
0121     % Centered L2-discrepancy
0122     part1=prod(1+0.5*abs(sampling-0.5)-0.5*(sampling-0.5).^2,2);
0123     opt=1+0.5*abs(samplingMC)+0.5*abs(samplingMBC);
0124     prodm=prod(opt,3);
0125     part2=sum(prodm(:));
0126     %     part2=0;
0127     %     for ii=1:nb_val
0128     %         for jj=1:nb_val
0129     %             tmp=1;
0130     %             for ll=1:nb_var
0131     %                 tmp=tmp*(1+0.5*abs(tirages(ii,ll)-0.5)+...
0132     %                     0.5*abs(tirages(jj,ll)-0.5));
0133     %             end
0134     %             part2=part2+tmp;
0135     %         end
0136     %     end
0137     
0138     
0139     discrepancy.CL2=(13/12)^np-2/ns*sum(part1)+...
0140         1/ns^2*part2;
0141     
0142     % Symetric L2-discrepancy
0143     part1=prod(1+2*sampling-2*sampling.^2,2);
0144     opt=1+0.5*abs(samplingMC)+0.5*abs(samplingMBC)-...
0145         0.5*abs(samplingMC-samplingMBC);
0146     prodm=prod(opt,3);
0147     part2=sum(prodm(:));
0148     %     part2=0;
0149     %     for ii=1:nb_val
0150     %         for jj=1:nb_val
0151     %             tmp=1;
0152     %             for ll=1:nb_var
0153     %                 tmp=tmp*(1+0.5*abs(tirages(ii,ll)-0.5)+...
0154     %                     0.5*abs(tirages(jj,ll)-0.5)-...
0155     %                     0.5*abs(tirages(ii,ll)-tirages(jj,ll)));
0156     %             end
0157     %             part2=part2+tmp;
0158     %         end
0159     %     end
0160     
0161     discrepancy.SL2=(4/3)^np-...
0162         2/ns*sum(part1)+...
0163         2^np/ns^2*part2;
0164     
0165     % Modified L2-discrepancy
0166     part1=prod(3-sampling.^2,2);
0167     opt=1-abs(samplingM-samplingMB);
0168     prodm=prod(opt,3);
0169     part2=sum(prodm(:));
0170     %     part2=0;
0171     %     for ii=1:nb_val
0172     %         for jj=1:nb_val
0173     %             tmp=1;
0174     %             for ll=1:nb_var
0175     %                 tmp=tmp*(1-abs(tirages(ii,ll)-tirages(jj,ll)));
0176     %             end
0177     %             part2=part2+tmp;
0178     %         end
0179     %     end
0180     discrepancy.ML2=(4/3)^np-...
0181         2^(1-np)/ns*sum(part1)+...
0182         1/ns^2*part2;
0183     
0184     % wrap around L2-discrepancy
0185     part1=abs(samplingM-samplingMB);
0186     opt=3/2-part1.*(1-part1);
0187     prodm=prod(opt,3);
0188     part2=sum(prodm(:));
0189     discrepancy.WL2=ns*(-(4/3)^np+...
0190         1/ns^2*part2);
