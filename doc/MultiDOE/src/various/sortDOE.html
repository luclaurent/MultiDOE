0154 function samplingSorted=nptp(sampling,lnorm,infoPt,type,doe)
0155 %number of sample point and variables
0156 [ns,np]=size(sampling);
0157 samplingSorted=zeros(ns,np);
0158 %list of the number of points
0159 listNnum=1:ns;
0160 %depending on the kind of initial point
0161 switch type
0162     case 'exist'
0163         %using an existing initial point
0164         numPtRef=infoPt;
0165         ptRef=sampling(numPtRef,:);
0166         samplingSorted(1,:)=ptRef;
0167         listNnum(numPtRef)=[];
0168         startFor=1;
0169     case 'don'
0170         %using a provided point
0171         ptRef=infoPt;
0172         startFor=0;
0173 end
0174 %for norming the points
0175 xNorm=doe.Xmax(:)'-doe.Xmin(:)';
0176 %sorting
0177 for ii=startFor:ns-1
0178     %compute the distance between the reference point and the others
0179     xDiff=ptRef(ones(1,ns-ii),:)-sampling(listNnum,:);
0180     xDiff=xDiff./xNorm(ones(1,ns-ii),:);
0181     distS=normp(xDiff,lnorm);
0182     %look for the minimal distance
0183     [~,IXDmin]=min(distS);
0184     numPtRef=listNnum(IXDmin(1));
0185     listNnum(IXDmin(1))=[];
0186     
0187     %new refernce point
0188     ptRef=sampling(numPtRef,:);
0189     samplingSorted(ii+1,:)=ptRef;
0190 end
