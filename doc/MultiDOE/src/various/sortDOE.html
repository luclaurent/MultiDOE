0025 function samplingSorted = sortDOE(sampling,doe)
0026 %for normalization
0027 Xmin=doe.Xmin;
0028 Xmax=doe.Xmax;

0031 %used norm
0032 lnorm=infoSort.lnorm;
0033 %number of sample points and design variables
0034 [ns,np]=size(sampling);

0040     sortOk=true;
0041     %various methods for sorting the sample points
0042     switch infoSort.type
0043         case {'v','variable'}
0044             %sorting wrt a variable
0045             if isfield(infoSort,'para')&&infoSort.para>0
0046                 if infoSort.para<=size(sampling,1)
0047                     [~,ind]=sort(sampling(:,infoSort.para));
0048                     samplingSorted=sampling(ind,:);
0049                 end
0050             end
0051         case {'nptp','normal_pt_to_pt'}
0052             % sorting with starting at a specific point (ptref or para) and by
0053             % looking from a point to another using a specific norm
0054             
0055             %initial point
0056             ptref=[];

0062             if isempty(ptref)

0064             else
0065                 ptref=(ptref-Xmin(:)')./Xmax(:)';
0066                 samplingSorted=nptp(sampling,lnorm,ptref,'don',doe);
0067             end
0068             
0069         case {'p','point'}
0070             %sorting by looking to the closest point to the barycenter of
0071             %the previous points
0072             
0073             %initial point
0074             ptref=[];
0075             if isfield(infoSort,'ptref');if ~isempty(infoSort.ptref);ptref=infoSort.ptref;end, end
0076             if isfield(infoSort,'para');numPtRef=infoSort.para;else numPtRef=1;end
0077             if numPtRef<1||numPtRef>np;numPtRef=1;end
0078

0082             else

0084                 samplingSorted=barypt(sampling,lnorm,ptref,'don',doe);
0085             end
0086             
0087         case {'c','center'}
0088             %same technique as 'p' but starting at the center of the

0099             %sorring
0100             samplingSorted=barypt(sampling,lnorm,centerPt,'don',doe);
0101         case {'sc','start_center'}
0102             %same as nptp but starting at the center of the design space
0103             centerPt=(Xmax(:)'+Xmin(:)')./2;
0104             %sorting
0105             samplingSorted=nptp(sampling,lnorm,centerPt,'don',doe);
0106             
0107         case {'sasc','sampling_start_center'}
0108             %same as nptp but starting at the the initial point is chosen as the

0158 function samplingSorted=nptp(sampling,lnorm,infoPt,type,doe)
0159 %number of sample point and variables
0160 [ns,np]=size(sampling);
0161 samplingSorted=zeros(ns,np);
0162 %list of the number of points
0163 listNnum=1:ns;
0164 %depending on the kind of initial point
0165 switch type
0166     case 'exist'
0167         %using an existing initial point
0168         numPtRef=infoPt;
0169         ptRef=sampling(numPtRef,:);
0170         samplingSorted(1,:)=ptRef;
0171         listNnum(numPtRef)=[];
0172         startFor=1;
0173     case 'don'
0174         %using a provided point
0175         ptRef=infoPt;
0176         startFor=0;
0177 end
0178 %for norming the points
0179 xNorm=doe.Xmax(:)'-doe.Xmin(:)';
0180 %sorting
0181 for ii=startFor:ns-1
0182     %compute the distance between the reference point and the others
0183     xDiff=ptRef(ones(1,ns-ii),:)-sampling(listNnum,:);
0184     xDiff=xDiff./xNorm(ones(1,ns-ii),:);
0185     distS=normp(xDiff,lnorm);
0186     %look for the minimal distance
0187     [~,IXDmin]=min(distS);
0188     numPtRef=listNnum(IXDmin(1));
0189     listNnum(IXDmin(1))=[];
0190     
0191     %new refernce point
0192     ptRef=sampling(numPtRef,:);
0193     samplingSorted(ii+1,:)=ptRef;
0194 end
0195 end
