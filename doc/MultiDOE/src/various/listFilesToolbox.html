0023 function listF=listFilesToolbox(dirT)
0024 listF={};
0025 %process all directories in the directory 'dirT'
0026 for ii=1:numel(dirT)
0027     %find files in directories
0028     fileDir=listFileDir(dirT{ii});
0029     %add name of the directory
0030     fun=@(x) sprintf('%s/%s',dirT{ii},x);
0031     fileDirOk=cellfun(fun,fileDir,'UniformOutput',false);
0032     %add to the whole list
0033     listF={listF{:},fileDirOk{:}};
0034 end
