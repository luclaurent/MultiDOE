0029 function listF=listFilesToolboxMultiDOE(dirT)
0030 listF={};
0031 %process all directories in the directory 'dirT'
0032 for ii=1:numel(dirT)
0033     %find files in directories
0034     fileDir=listFileDir(dirT{ii});
0035     %add name of the directory
0036     fun=@(x) sprintf('%s/%s',dirT{ii},x);
0037     fileDirOk=cellfun(fun,fileDir,'UniformOutput',false);
0038     %add to the whole list
0039     listF={listF{:},fileDirOk{:}};
0040 end
