0025 function listF=listFilesToolbox(dirT)
0026 listF={};
0027 %process all directories in the directory 'dirT'
0028 for ii=1:numel(dirT)
0029     %find files in directories
0030     fileDir=listFileDir(dirT{ii});
0031     %add name of the directory
0032     fun=@(x) sprintf('%s/%s',dirT{ii},x);
0033     fileDirOk=cellfun(fun,fileDir,'UniformOutput',false);
0034     %add to the whole list
0035     listF={listF{:},fileDirOk{:}};
0036 end
0037 %add the root directory of the Toolbox
0038 fileDirOk=listFileDir('.');
0039 %add to the whole list
0040 listF={listF{:},fileDirOk{:}};
