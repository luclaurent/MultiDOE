0027 %% INPUT:
0028 %    - Xmin,Xmax: min and max bounds of the design space

0040 %%initialize options
0041 % storing directory
0042 folderStore='tmpDOE/LHS_R';
0043 %number of initial sampling
0044 nbInitSampling=0;
0045 %name of the R script file
0046 nameScript='mmlhs_R_';
0047 extScript='.r';
0048 %name of the R data file
0049 nameDataR='dataR_';
0050 extDataR='.dat';
0051 %pause time after executing R
0052 timePause=0;

0054 %number of permutations "Columnwise Pairwise"
0055 nbPerm=numel(Xmin);
0056 %stoping criterion
0057 critStop=1e-2;
0058 
0059 %building DOE
0060 if nargin==3
0061     
0062     % load dimensions (number of variables and sample points)
0063     np=numel(Xmin);
0064     %full name of the R script file
0065     nameScript=[nameScript num2str(np) '_' num2str(ns) extScript];
0066     %full name of the R data file
0067     nameDataR=[nameDataR num2str(np) '_' num2str(ns) extDataR];
0068     
0069     %create storing folder if not existing
0070     if exist(folderStore,'dir')~=7
0071         mkdir(folderStore);
0072     end
0073     
0074     %%write R script
0075     textInit=['a<-maximinLHS(' num2str(ns) ',' num2str(np) ','...
0076         num2str(nbPerm) ',' num2str(critStop) ')\n'];

0079     %load LHS library
0080     loadLHS='library(lhs)\n';

0083

0094     %write storage procedure

0109     newSampling=[];
0110     
0111     %enrichment procedure
0112 elseif nargin==5
0113

0118     %new sampling

0120     newSampling=A(ind,:).*repmat(Xmax(:)'-Xmin(:)',nbInfill,1)+repmat(Xmin(:)',nbInfill,1);
0121     %all sample points
