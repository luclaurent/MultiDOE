0027 %% INPUT:
0028 %    - Xmin,Xmax: min and max bounds of the design space

0037 %path declaration for R software
0038 setenv('DYLD_LIBRARY_PATH','/usr/local/bin/');

0040 %%initialize options
0041 % storing directory
0042 folderStore='tmpDOE/LHS_R';
0043 %number of initial sampling
0044 nbInitSampling=0;
0045 %name of the R script file
0046 nameScript='olhs_R_';
0047 extScript='.r';
0048 %name of the R data file
0049 nameDataR='dataR_';
0050 extDataR='.dat';
0051 %pause time after executing R
0052 timePause=0;
0053 %number of permuations "Columnwise Pairwise"
0054 nbPerm=numel(Xmin);
0055 %stoping criterion
0056 critStop=1e-2;
0057 
0058 %building DOE
0059 if nargin==3
0060     
0061     % load dimensions (number of variables and sample points)
0062     np=numel(Xmin);
0063     %full name of the R script file
0064     nameScript=[nameScript num2str(np) '_' num2str(ns) extScript];
0065     %fulle name of the R data file
0066     nameDataR=[nameDataR num2str(np) '_' num2str(ns) extDataR];
0067     
0068     %create storing folder if not existing
0069     if exist(folderStore,'dir')~=7
0070         mkdir(folderStore);
0071     end
0072     
0073     %%write R script
0074     textInit=['a<-optimumLHS(' num2str(ns) ',' num2str(np) ','...
0075         num2str(nbPerm) ',' num2str(critStop) ')\n'];
0076     %infill procedure
0077     texttextInfill_enrich=['a<-optAugmentLHS(a,1,4)\n'];
0078     %load LHS library
0079     loadLHS='library(lhs)\n';
0080     %store sampling
0081     storeSampling=['write.table(a,file="' nameDataR '",row.names=FALSE,col.names=FALSE)'];
0082     
0083     %create and open script file
0084     fid=fopen([folderStore '/' nameScript],'w','n','UTF-8');
0085     %write loading of the library

0088     fprintf(fid,textInit);
0089     %write enrichment

0091         fprintf(fid,texttextInfill_enrich);
0092     end
0093     %write storage procedure

0095     %close file
0096     fclose(fid);
0097     %%execute R (R must be installed)
0098     %check if available
0099     [rOk,TBXOk]=checkR;

0103     end
0104     %read data file
0105     A=load([folderStore '/' nameDataR]);

0107     sampling=A(1:ns,:).*repmat(Xmax(:)'-Xmin(:)',ns,1)+repmat(Xmin(:)',ns,1);
0108     newSampling=[];
0109     
0110     %enrichment procedure
0111 elseif nargin==5
0112     %number of sample points in the initial sampling

0114     %read data file
0115     A=load([folderStore '/' nameDataR]);
0116     %new sampling
0117     ind=nsOld+1:nsOld+nbInfill;
0118     newSampling=A(ind,:).*repmat(Xmax(:)'-Xmin(:)',nbInfill,1)+repmat(Xmin(:)',nbInfill,1);
0119     %all sample points
0120     sampling=[oldSampling;newSampling];
