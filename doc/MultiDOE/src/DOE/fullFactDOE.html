0033 sampling=zeros(nsTot,np);

0041 %% compute values of the sampling matrix
0042 % along the varibales

0048         nbTerPre=1;

0053         temp1=[temp1;temp];
0054     end
0055     temp2=repmat(temp1,prod(ns(ii+1:end)),1);
