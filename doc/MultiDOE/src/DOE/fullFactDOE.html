0018 %% Building a full factorial doe with nD variables
0019 % L. LAURENT -- 07/10/2011 -- luc.laurent@lecnam.net
0020 
0021 function sampling=fullFactDOE(ns,Xmin,Xmax)
0022 
0023 %number of variables
0024 np=numel(Xmin);
0025 %reodering
0026 if length(ns)==1
0027     ns=ns*ones(1,np);
0028 elseif length(ns)~=1&&length(ns)~=np
0029     error(['wrong definition of the number of required sample points (',mfilename,')']);
0030 end
0031 
0032 %initialize matrix for storing sample points
0033 nsTot=prod(ns);
0034 sampling=zeros(nsTot,np);
0035 
0036 %values for each variables
0037 valVar=cell(np,1);
0038 for ii=1:np
0039     valVar{ii}=linspace(Xmin(ii),Xmax(ii),ns(ii));
0040 end
0041 
0042 %% compute values of the sampling matrix
0043 % along the varibales
0044 temp1=[];
0045 for ii=1:np
0046     if ii>1
0047         nbTerPre=prod(ns(1:ii-1));
0048     else
0049         nbTerPre=1;
0050     end
0051     %along values per variables
0052     for jj=1:length(valVar{ii})
0053         temp=repmat(valVar{ii}(jj),nbTerPre,1);
0054         temp1=[temp1;temp];
0055     end
0056     temp2=repmat(temp1,prod(ns(ii+1:end)),1);
0057     temp1=[];
0058     sampling(:,ii)=temp2;
0059 end
0060 end
